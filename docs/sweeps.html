if 'WANDB_API_KEY' not in os.environ:
    if os.environ['WANDB_MODE'] == 'online':
        from getpass import getpass
        secret = getpass('Enter WandB API Key: ')
        os.environ['WANDB_API_KEY'] = secret
    else:
